def generate_truth_table(num_vars, func):
    """Generate truth table for a boolean function"""
    print("Truth Table:")
    header = " ".join([f"x{i}" for i in range(num_vars)] + ["f"])
    print(header)
    print("-" * len(header))

    for i in range(2 ** num_vars):
        # Convert number to binary inputs
        inputs = [(i >> j) & 1 for j in range(num_vars-1, -1, -1)]
        result = func(*inputs)
        row = " ".join(str(x) for x in inputs + [result])
        print(row)

Building Gates from NAND:

def evaluate_boolean_expr(expr, values):
    """Evaluate a boolean expression given variable values"""
    # Replace variables with their values
    for var, val in values.items():
        expr = expr.replace(var, str(val))

    # Replace boolean operators
    expr = expr.replace('Â·', ' and ')
    expr = expr.replace('+', ' or ')
    expr = expr.replace("'", ' not ')

    return eval(expr)

class LogicGate:
    """Simulate basic logic gates"""

    @staticmethod
    def AND(a, b):
        return a & b

    @staticmethod
    def OR(a, b):
        return a | b

    @staticmethod
    def NOT(a):
        return 1 - a  # or ~a & 1

    @staticmethod
    def NAND(a, b):
        return LogicGate.NOT(LogicGate.AND(a, b))

    @staticmethod
    def NOR(a, b):
        return LogicGate.NOT(LogicGate.OR(a, b))

    @staticmethod
    def XOR(a, b):
        return a ^ b

    @staticmethod
    def XNOR(a, b):
        return LogicGate.NOT(LogicGate.XOR(a, b))

ğŸ’» Half Adder & Full Adder

def half_adder(a, b):
    """Returns (sum, carry)"""
    sum_bit = a ^ b  # XOR
    carry = a & b    # AND
    return sum_bit, carry

def full_adder(a, b, cin):
    """Returns (sum, carry_out)"""
    sum1, carry1 = half_adder(a, b)
    sum_final, carry2 = half_adder(sum1, cin)
    carry_out = carry1 | carry2
    return sum_final, carry_out

# Test
print("Half Adder:")
for a in [0, 1]:
    for b in [0, 1]:
        s, c = half_adder(a, b)
        print(f"{a} + {b} = {s}, carry={c}")

print("\nFull Adder:")
for a in [0, 1]:
    for b in [0, 1]:
        for cin in [0, 1]:
            s, cout = full_adder(a, b, cin)
            print(f"{a} + {b} + {cin} = {s}, carry={cout}")

def ripple_carry_adder(a, b):
    """Add two 4-bit numbers represented as lists"""
    result = []
    carry = 0

    for i in range(len(a)):
        sum_bit, carry = full_adder(a[i], b[i], carry)
        result.append(sum_bit)

    return result, carry

# Example: 5 + 6 = 11

b = [0, 1, 1, 0]  # 6 in binary
